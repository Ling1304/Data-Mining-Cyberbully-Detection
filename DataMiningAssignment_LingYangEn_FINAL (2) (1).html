def remove_special_characters(input_text):
    # Using regex to substitute all non-alphanumeric characters with an empty string
    cleaned_text = re.sub(r'[^a-zA-Z0-9\s]', '', input_text)
    return cleaned_text

df['Comment'] =  df['Comment'].apply(lambda x: remove_special_characters(x))

def remove_numbers(input_text):
    # Using regex to substitute all numeric characters with an empty string
    cleaned_text = re.sub(r'\d', '', input_text)
    return cleaned_text

df['Comment'] =  df['Comment'].apply(lambda x: remove_numbers(x))
